          function isPrime(n) {
          for (let i = 2, s = Math.sqrt(n); i <= s; i++) { if (n % i==0) return false; } return n> 1;
            }
        
            function step(g, m, n) {
            const arr = [];
            for (let i = m; i <= n; i++) { 
              if (isPrime(i)) arr.push(i); 
            } 
            for (let i=0; i < arr.length - 1; i++) { 
              for (let j=1; j < arr.length - i; j++) { 
                if (arr[i + j] - arr[i]==g) return [arr[i], arr[i + j]]; 
              } } 
              return null; 
            }
